package main

import (
	"fmt"
	"sync"
)

var once sync.Once

func initialize() {
	fmt.Println("This will not be repeated no matter how many times you call this function using once.Do")
}

func main() {
	var wg sync.WaitGroup
	for i := range 5 {
		wg.Go(func() {
			fmt.Println("Goroutine ", i)
			once.Do(initialize)
		})
	}

	wg.Wait()
}
